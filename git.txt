git init
git status
git add . : add all
git add git.txt : add selected file
git commit -m "Comment" -m "Discription" :Commit file
git commit -am "Comment" : Add and commit at same time
git push -u origin master : push commited file to remote machine

git branch : List branches
git checkout -b feature-1 : Create new branch with name feature-1
git checkout master : switch between branches

git diff : check difference 
git merge feature1 : merge branch name "feature1" with other branch
git branch -d nameofbranch : delete branch
git pull -u origin master
